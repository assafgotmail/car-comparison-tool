        function getFeatureDetailHTML(label, status) {

            let icon = '';

            let textColor = 'text-gray-800';


            if (status.available === true) {

                icon = '<i class="ph-duotone ph-check-circle text-2xl text-green-500"></i>';

            } else if (status.available === 'trim') {

                icon = '<i class="ph-duotone ph-warning-circle text-2xl text-yellow-500"></i>';

                textColor = 'text-gray-600';

            } else {

                icon = '<i class="ph-duotone ph-x-circle text-2xl text-red-500"></i>';

                textColor = 'text-gray-600 line-through';

            }


            return `

                <li class="flex items-center gap-3">

                    ${icon}

                    <div>

                        <span class="font-medium ${textColor}">${label}</span>

                        <span class="block text-xs text-gray-500">${status.note || (status.available ? 'Standard' : 'Not available')}</span>

                    </div>

                </li>

            `;

        }

        function openModal(carId) {

            const carIndex = state.cars.findIndex(c => c.id === carId);

            if (carIndex === -1) return;

            

            state.currentCarIndex = carIndex;

            const car = state.cars[carIndex];

            populateModal(car);

            

            modal.classList.remove('hidden');

            setTimeout(() => {

                modalContainer.classList.remove('scale-95', 'opacity-0');

                modalContainer.classList.add('scale-100', 'opacity-100');

            }, 10);

        }


        function closeModal() {

            modalContainer.classList.add('scale-95', 'opacity-0');

            modalContainer.classList.remove('scale-100', 'opacity-100');

            setTimeout(() => {

                modal.classList.add('hidden');

            }, 300);

        }


        function showNextPrevCar(direction) {

            let newIndex = state.currentCarIndex + direction;

            if (newIndex < 0) newIndex = state.cars.length - 1;

            if (newIndex >= state.cars.length) newIndex = 0;

            

            // Find the ID from the *filtered* list

            const newCarId = state.cars[newIndex].id;

            // Find the actual car object from the *global* list to populate

            const car = carData.find(c => c.id === newCarId);

            if (car) {

                state.currentCarIndex = newIndex; // Update index relative to filtered list

                populateModal(car);

            }

        }

        function getSortValue(car, sortKey) {

            if (featureDefinitions.map(f => f.id).includes(sortKey)) {

                const feature = car.features[sortKey];

                if (!feature) return 1;

                if (feature.available === true) return 3;

                if (feature.available === 'trim') return 2;

                return 1; // false or undefined

            }

            if (sortKey === 'price' || sortKey === 'range') {

                return car[sortKey] || 0; // Treat null/undefined as 0 for sorting

            }

            return car[sortKey];

        }


        function applyFiltersAndSort() {

            let filteredCars = [...carData];


            if (state.filters.powertrain !== 'all') {

                filteredCars = filteredCars.filter(car => car.powertrain === state.filters.powertrain);

            }

            if (state.filters.price !== 'all') {

                const maxPrice = parseInt(state.filters.price, 10);

                filteredCars = filteredCars.filter(car => car.price <= maxPrice);

            }

            if (state.filters.features.length > 0) {

                filteredCars = filteredCars.filter(car => {

                    return state.filters.features.every(featureKey => {

                        const feature = car.features[featureKey];

                        return feature && feature.available; // 'true' or 'trim'

                    });

                });

            }


            filteredCars.sort((a, b) => {

                let valA = getSortValue(a, state.sortBy);

                let valB = getSortValue(b, state.sortBy);


                let modifier = state.sortOrder === 'asc' ? 1 : -1;

                

                if (valA < valB) return -1 * modifier;

                if (valA > valB) return 1 * modifier;

                return 0;

            });


            state.cars = filteredCars; // Update the filtered list

            renderTable();

            updateSortIcons();

        }

        async function fetchWithBackoff(apiUrl, payload, retries = 3, delay = 1000) {

            try {

                const response = await fetch(apiUrl, {

                    method: 'POST',

                    headers: { 'Content-Type': 'application/json' },

                    body: JSON.stringify(payload)

                });

                if (!response.ok) throw new Error(`HTTP error! status: ${response.status}`);

                return await response.json();

            } catch (error) {

                if (retries > 0) {

                    await new Promise(resolve => setTimeout(resolve, delay));

                    return fetchWithBackoff(apiUrl, payload, retries - 1, delay * 2);

                } else {

                    console.error("API call failed after retries:", error);

                    throw error;

                }

            }

        }

        document.addEventListener('DOMContentLoaded', () => {

            populateAddCarFormFeatures();

            applyFiltersAndSort();


            // Filter listeners

            filterPowertrain.addEventListener('change', (e) => {

                state.filters.powertrain = e.target.value;

                applyFiltersAndSort();

            });

            filterPrice.addEventListener('change', (e) => {

                state.filters.price = e.target.value;

                applyFiltersAndSort();

            });

            filterFeatureCheckboxes.forEach(checkbox => {

                checkbox.addEventListener('change', () => {

                    state.filters.features = 

                        Array.from(filterFeatureCheckboxes)

                             .filter(cb => cb.checked)

                             .map(cb => cb.value);

                    applyFiltersAndSort();

                });

            });


            // Sort listeners

            sortableHeaders.forEach(header => {

                header.addEventListener('click', () => {

                    const sortKey = header.dataset.sort;

                    if (!sortKey) return;

                    

                    if (state.sortBy === sortKey) {

                        state.sortOrder = state.sortOrder === 'asc' ? 'desc' : 'asc';

                    } else {

                        state.sortBy = sortKey;

                        const defaultDesc = ['price', 'range', ...featureDefinitions.map(f => f.id)];

                        state.sortOrder = defaultDesc.includes(sortKey) ? 'desc' : 'asc';

                    }

                    applyFiltersAndSort();

                });

            });


            // Details Modal Listeners

            tableBody.addEventListener('click', (e) => {

                if (e.target.closest('a')) return;

                const row = e.target.closest('tr');

                if (row && row.dataset.carId) {

                    openModal(parseInt(row.dataset.carId, 10));

                }

            });


            closeModalBtn.addEventListener('click', closeModal);

            prevCarBtn.addEventListener('click', () => showNextPrevCar(-1));

            nextCarBtn.addEventListener('click', () => showNextPrevCar(1));

            

            // 3. Edit/Delete listeners

            modalEditBtn.addEventListener('click', () => {

                const car = carData.find(c => c.id === state.currentCarId);

                if (car) {

                    closeModal();

                    openEditCarModal(car);

                }

            });

            modalDeleteBtn.addEventListener('click', openDeleteConfirmModal);


            // 1. Remove listener for Review Summary


            // Add Car Modal Listeners

            openAddCarBtn.addEventListener('click', openAddCarModal);

            closeAddCarBtn.addEventListener('click', closeAddCarModal);

            cancelAddCarBtn.addEventListener('click', closeAddCarModal);

            addCarForm.addEventListener('submit', handleAddCarSubmit);


            // 4. Auto-fill listener

            findCarInfoBtn.addEventListener('click', handleFindCarInfo);


            // Image Drop Zone Listeners

            imageDropZone.addEventListener('click', () => {

                imageFileInput.click();

            });

            imageFileInput.addEventListener('change', (e) => {

                if (e.target.files.length > 0) {

                    handleImageFile(e.target.files[0]);

                }

            });

            imageDropZone.addEventListener('dragover', (e) => {

                e.preventDefault();

                imageDropZone.classList.add('drop-zone-active');

            });

            imageDropZone.addEventListener('dragleave', (e) => {

                e.preventDefault();

                imageDropZone.classList.remove('drop-zone-active');

            });

            imageDropZone.addEventListener('drop', (e) => {

                e.preventDefault();

                imageDropZone.classList.remove('drop-zone-active');

                if (e.dataTransfer.files.length > 0) {

                    handleImageFile(e.dataTransfer.files[0]);

                }

            });

            // Listen for pastes on the URL input field

            addImageUrlInput.addEventListener('paste', (e) => {

                if (e.clipboardData.files.length > 0) {

                    e.preventDefault(); // Prevent pasting file path text

                    handleImageFile(e.clipboardData.files[0]);

                }

            });


            // Delete Confirm Modal Listeners

            cancelDeleteBtn.addEventListener('click', closeDeleteConfirmModal);

            confirmDeleteBtn.addEventListener('click', handleDeleteCar);


            // Global Listeners

            window.addEventListener('keydown', (e) => {

                if (e.key === 'Escape') {

                    if (!modal.classList.contains('hidden')) closeModal();

                    if (!addCarModal.classList.contains('hidden')) closeAddCarModal();

                    if (!deleteConfirmModal.classList.contains('hidden')) closeDeleteConfirmModal();

                }

            });

            modal.addEventListener('click', (e) => { if (e.target === modal) closeModal(); });

            addCarModal.addEventListener('click', (e) => { if (e.target === addCarModal) closeAddCarModal(); });

            deleteConfirmModal.addEventListener('click', (e) => { if (e.target === deleteConfirmModal) closeDeleteConfirmModal(); });

        });


    </script>
